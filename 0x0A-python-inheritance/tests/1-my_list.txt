This file test if the MyList class can sort a list that it inherits

::

        >>> MyList = __import__('1-my_list').MyList
        >>> dlist = MyList()
	>>> dlist += [-1, 2, -2, 0, -15]
        >>> dlist.print_sorted()
        [-15, -2, -1, 0, 2]
        >>> print(dlist)
        [-1, 2, -2, 0, -15]
        >>> dlist = MyList()
	>>> dlist += [-1, -33, -77, -6, -25]
        >>> dlist.print_sorted()
        [-77, -33, -25, -6, -1]
        >>> print(dlist)
        [-1, -33, -77, -6, -25]
        >>> dlist = MyList()
	>>> dlist += [1.2, 2.3, 7.7, 0.3, 15.2]
        >>> dlist.print_sorted()
        [0.3, 1.2, 2.3, 7.7, 15.2]
        >>> print(dlist)
        [1.2, 2.3, 7.7, 0.3, 15.2]
	>>> dlist = MyList()
	>>> dlist += [1, 3, 1, 2, 3]
	>>> dlist.print_sorted()
	[1, 1, 2, 3, 3]
	>>> print(dlist)
	[1, 3, 1, 2, 3]
	>>> dlist = MyList()
	>>> dlist.print_sorted()
	[]
	>>> print(dlist)
	[]
	>>> dlist = MyList()
	>>> dlist += [-1.3, -14.27, -2.33, -10.4, -5.29]
	>>> dlist.print_sorted()
	[-14.27, -10.4, -5.29, -2.33, -1.3]
	>>> print(dlist)
	[-1.3, -14.27, -2.33, -10.4, -5.29]
	>>> dlist = MyList()
	>>> dlist += [0, 0, 0, 0, 0]
	>>> dlist.print_sorted()
	[0, 0, 0, 0, 0]
	>>> print(dlist)
	[0, 0, 0, 0, 0]
